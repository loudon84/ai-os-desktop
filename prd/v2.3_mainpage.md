```ts id="uslrx9"
import type { BrowserWindow } from "electron";
import { ShellViewEvents } from "../../shared/shell/shell-view-contract";
import { viewEventBus } from "./views/view-events";

export function bindShellViewEventForwarder(mainWindow: BrowserWindow): () => void {
  const send = (channel: string, payload: unknown): void => {
    if (!mainWindow.isDestroyed()) {
      mainWindow.webContents.send(channel, payload);
    }
  };

  const onMetadataChanged = (payload: unknown): void => {
    send(ShellViewEvents.METADATA_CHANGED, payload);
  };

  const onLoadFailed = (payload: unknown): void => {
    send(ShellViewEvents.LOAD_FAILED, payload);
  };

  const onCrashed = (payload: unknown): void => {
    send(ShellViewEvents.CRASHED, payload);
  };

  viewEventBus.on("view:metadata-changed", onMetadataChanged as never);
  viewEventBus.on("view:load-failed", onLoadFailed as never);
  viewEventBus.on("view:crashed", onCrashed as never);

  return () => {
    viewEventBus.off("view:metadata-changed", onMetadataChanged as never);
    viewEventBus.off("view:load-failed", onLoadFailed as never);
    viewEventBus.off("view:crashed", onCrashed as never);
  };
}
```

```ts id="gdwx4c"
import { useEffect, useState } from "react";
import type { ShellViewSnapshot } from "../../../../shared/shell/shell-view-contract";

export function useShellViewMetadata(): Record<string, ShellViewSnapshot> {
  const [metadataById, setMetadataById] = useState<Record<string, ShellViewSnapshot>>({});

  useEffect(() => {
    let disposed = false;

    void window.shellView.getAll().then((items) => {
      if (disposed) return;
      setMetadataById(Object.fromEntries(items.map((item) => [item.id, item])));
    });

    const offMetadata = window.shellView.onMetadataChanged((event) => {
      setMetadataById((prev) => ({
        ...prev,
        [event.snapshot.id]: event.snapshot,
      }));
    });

    const offLoadFailed = window.shellView.onLoadFailed((event) => {
      setMetadataById((prev) => {
        const current = prev[event.id];
        if (!current) return prev;

        return {
          ...prev,
          [event.id]: {
            ...current,
            errorCode: event.errorCode,
            errorDescription: event.errorDescription,
            updatedAt: Date.now(),
          },
        };
      });
    });

    const offCrashed = window.shellView.onCrashed((event) => {
      setMetadataById((prev) => {
        const current = prev[event.id];
        if (!current) return prev;

        return {
          ...prev,
          [event.id]: {
            ...current,
            crashed: true,
            crashedReason: event.reason,
            crashedExitCode: event.exitCode,
            updatedAt: Date.now(),
          },
        };
      });
    });

    return () => {
      disposed = true;
      offMetadata();
      offLoadFailed();
      offCrashed();
    };
  }, []);

  return metadataById;
}
```

## 10.6 KeepAlive 内存控制

长期保活会增加 React 内存占用，需要加边界。

新增 hook：

```txt id="t9y9ej"
src/renderer/src/components/layout/useKeepAliveRegistry.ts
```

第一版只记录访问状态：

```ts id="mq8gdb"
export interface KeepAliveEntry {
  view: string;
  mountedAt: number;
  lastActiveAt: number;
}

export function useKeepAliveRegistry(activeView: string) {
  const [entries, setEntries] = useState<Record<string, KeepAliveEntry>>({});

  useEffect(() => {
    setEntries((prev) => ({
      ...prev,
      [activeView]: prev[activeView]
        ? { ...prev[activeView], lastActiveAt: Date.now() }
        : {
            view: activeView,
            mountedAt: Date.now(),
            lastActiveAt: Date.now(),
          },
    }));
  }, [activeView]);

  return entries;
}
```

```txt id="o4eyjs"

2. LRU 卸载低频页面
3. settings / gateway / chat 永久保活
```

本阶段先不做 LRU，避免引入状态复杂度。

# 12. Commit 10：Resize / bounds 稳定化

当前 `ShellViewManager.handleWindowResize()` 仍是占位逻辑，注释说明应该存储原始 layout。

补充：

```ts id="vpwbxr"
private lastActivationLayoutById = new Map<string, ShellViewLayout>();
private lastActivationBoundsById = new Map<string, ShellViewBounds>();
```

在 `activateView()` 中记录：

```ts id="e99xg9"
if (boundsOrLayout) {
  if (this.isLayout(boundsOrLayout)) {
    this.lastActivationLayoutById.set(id, boundsOrLayout);
    this.lastActivationBoundsById.delete(id);
  } else {
    this.lastActivationBoundsById.set(id, boundsOrLayout);
    this.lastActivationLayoutById.delete(id);
  }
}
```

resize 时只重算 layout 型 view：

```ts id="ab0yzi"
private handleWindowResize(): void {
  const winBounds = this.mainWindow.getBounds();
  const calc = new LayoutCalculator(winBounds.width, winBounds.height);

  for (const ids of this.activeViewsByLayer.values()) {
    for (const id of ids) {
      const view = this.views.get(id);
      if (!view || !view.isActive()) continue;

      const layout = this.lastActivationLayoutById.get(id);
      if (!layout) continue;

      view.updateBounds(calc.calculate(layout));
    }
  }
}
```

WebContentsHost 仍然优先用 ResizeObserver 同步真实 DOM bounds。当前 `WebContentsHost` 已经用 `ResizeObserver + window.resize` 调 `window.shellView.setBounds(layerId, bounds)`。

# 13. Commit 11：Security hardening

当前 `LayoutCalculator.evaluateCalc()` 使用 `new Function`。

替换成有限表达式解析，只支持加减：

```ts id="k5gz9b"
private evaluateCalc(expression: string, baseSize: number): number {
  const normalized = expression
    .replace(/(\d+(?:\.\d+)?)%/g, (_match, percent) => {
      return String((baseSize * parseFloat(percent)) / 100);
    })
    .replace(/(\d+(?:\.\d+)?)px/g, "$1")
    .replace(/\s+/g, "");

  const tokens = normalized.match(/[+-]?\d+(?:\.\d+)?/g);
  if (!tokens) return 0;

  const result = tokens.reduce((sum, token) => sum + Number(token), 0);
  return Math.max(0, Math.floor(result));
}
```

支持：

```txt id="d9wzoo"
calc(100% - 40px)
calc(50% + 100px)
calc(100% - 40px - 24px)
```

不支持乘除。桌面布局阶段不需要乘除。
