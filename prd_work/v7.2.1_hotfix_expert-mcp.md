## 2. 当前问题

### 2.1 现象

当前 `本地 Hermes -> 专家` 页面展示内容仍是远端 skill/tool 明细，例如：

```text
b2b-contact-finder
customer-profiling
electronics-industry-news-digest
enterprise-risk-analysis
industry-search
manufacturer-profiling
semiconductor-marketing-copy
technology-convergence-analysis
```

页面标题虽然是“专家”，但数据实际上仍来自旧专家目录或 skill 列表聚合结果。

---

### 2.2 已确认的当前调用链

当前 Experts 页面链路为：

```text
Renderer
  HermesExpertsPage
    ↓
  HermesExpertsProvider.loadFromApi()
    ↓
  window.hermesExperts.listExpertCatalog()
    ↓
Preload
  ipcRenderer.invoke("hermes-experts:list-catalog")
    ↓
Main Process
  registerHermesExpertsIpc()
    ↓
  listExpertCatalog()
    ↓
HTTP
  GET {backendBaseUrl}/api/v1/hermes/experts?page=1&pageSize=20
```

当前 Expert Teams 页面链路为：

```text
Renderer
  window.hermesExperts.listExpertTeams()
    ↓
Preload
  ipcRenderer.invoke("hermes-experts:list-teams")
    ↓
Main Process
  listExpertTeams()
    ↓
HTTP
  GET {backendBaseUrl}/api/v1/hermes/expert-teams
```

当前本地 MCP Gateway 代理链路为：

```text
Renderer / Hermes Local MCP
  ↓
http://127.0.0.1:48742/mcp?profile=default
  ↓
Main Process local MCP proxy
  ↓
{backendBaseUrl}/api/v1/hermes/mcp
```

该链路用于 MCP Gateway 页面、本地 Hermes 注册 MCP，不应作为 Experts 页面远端专家目录的主链路。

---

## 3. 根因分析

### 3.1 根因一：Experts Catalog 主路径错误

`src/main/hermes-experts/expert-catalog-client.ts` 当前仍以旧 REST API 为主路径：

```text
GET /api/v1/hermes/experts
GET /api/v1/hermes/expert-teams
```

但 nodeskclaw 新 Expert MCP 合约要求：

```text
POST /api/v1/expert/mcp
method = tools/list
```

因此 Experts 页面拿到的是旧接口返回内容，无法按照 `annotations.kind` 判断：

```text
expert
expert_team
expert_skill
expert_team_skill
```

---

### 3.2 根因二：Expert / Skill / Tool 对象混用

当前页面将后端返回的条目统一渲染成 `HermesExpert` 卡片，但没有强制校验：

```text
annotations.kind = expert
```

导致 skill 明细被误展示为专家。

---

### 3.3 根因三：旧 MCP Gateway 与新 Expert MCP Gateway 未分层

当前已有两类 MCP 入口：

```text
1. /api/v1/hermes/mcp
   用于 Hermes MCP Skill Gateway、Local Hermes MCP 注册、MCP Gateway 页面调试。

2. /api/v1/expert/mcp
   用于 Desktop Experts / Expert Teams 页面发现和召唤远端专家。
```

当前代码没有建立独立的 `ExpertMcpClient`，导致 Experts 页面仍沿用旧链路。

---

## 4. 版本目标

### 4.1 产品目标

```text
1. Experts 页面只展示真正的远端专家。
2. Expert Teams 页面只展示真正的专家团队。

7. 旧 /api/v1/hermes/experts 不再作为 Experts 页面主链路。
```


## FR-001：新增 Expert MCP endpoint resolver

### 目标

建立独立的 Expert MCP endpoint 解析逻辑，不复用 Hermes MCP Skill Gateway endpoint。

### 新增文件

```text
src/main/hermes-experts/expert-mcp-endpoint.ts
```

### 实现

```ts
export function resolveExpertMcpBaseUrl(): string {
  const backend = resolveBackendBaseUrl();
  if (!backend) return "";
  return `${backend.replace(/\/+$/, "")}/api/v1/expert`;
}

export function resolveExpertMcpRootUrl(): string {
  return `${resolveExpertMcpBaseUrl()}/mcp`;
}

export function resolveExpertMcpSlugUrl(slug: string): string {
  return `${resolveExpertMcpBaseUrl()}/mcp/${encodeURIComponent(slug)}`;
}

export function resolveExpertHealthUrl(): string {
  return `${resolveExpertMcpBaseUrl()}/health`;
}
```

### 验收

```text
1. Expert MCP endpoint 固定为 /api/v1/expert。
2. 不读取 mcpEndpointPath。
3. 不使用 /api/v1/hermes/mcp。
4. 不使用本地 proxy http://127.0.0.1:48742/mcp。
```

```text
  ↓
```


```ts
import { resolveBackendBaseUrl } from "../mcp-skill-gateway-runtime/mcp-skill-gateway-config";

function joinUrl(base: string, path: string): string {
  return `${base.replace(/\/+$/, "")}${path.startsWith("/") ? path : `/${path}`}`;
}

export function resolveExpertMcpBaseUrl(): string {
  const backend = resolveBackendBaseUrl();
  if (!backend) return "";
  return joinUrl(backend, "/api/v1/expert");
}

export function resolveExpertMcpRootUrl(): string {
  const base = resolveExpertMcpBaseUrl();
  return base ? joinUrl(base, "/mcp") : "";
}

export function resolveExpertMcpSlugUrl(slug: string): string {
  const base = resolveExpertMcpBaseUrl();
  return base ? joinUrl(base, `/mcp/${encodeURIComponent(slug)}`) : "";
}

export function resolveExpertHealthUrl(): string {
  const base = resolveExpertMcpBaseUrl();
  return base ? joinUrl(base, "/health") : "";
}
```

### 12.2 ExpertMcpClient.listCatalog

```ts
export async function listExpertMcpCatalog(): Promise<RemoteCatalogItem[]> {
  const url = resolveExpertMcpRootUrl();
  if (!url) {
    throw new HermesExpertsError(
      "EXPERT_MCP_ENDPOINT_NOT_CONFIGURED",
      "Expert MCP endpoint is not configured"
    );
  }

  const result = await postExpertJsonRpc<ToolsListResult>(url, {
    jsonrpc: "2.0",
    id: `expert-catalog-${Date.now()}`,
    method: "tools/list",
    params: {}
  });

  return result.tools
    .map(mapCatalogTool)
    .filter((item): item is RemoteCatalogItem => item != null);
}
```

### 12.3 mapCatalogTool

```ts
export function mapCatalogTool(tool: ExpertMcpToolDescriptor): RemoteCatalogItem | null {
  const annotations = tool.annotations ?? {};
  const kind = annotations.kind;

  if (kind !== "expert" && kind !== "expert_team") {
    return null;
  }

  const slug =
    readString(annotations.slug) ||
    readString(annotations.expert_slug) ||
    readString(annotations.team_slug) ||
    tool.name;

  if (!slug) return null;

  return {
    slug,
    kind,
    remoteToolName: tool.name,
    displayName:
      readString(annotations.displayName) ||
      readString(annotations.display_name) ||
      tool.name,
    description: tool.description ?? "",
    category: readString(annotations.category) || "general",
    tags: readStringArray(annotations.tags),
    status: normalizeCatalogStatus(annotations.status),
    publicSkillCount: readNumber(annotations.publicSkillCount) ?? 0,
    callableSkillCount: readNumber(annotations.callableSkillCount) ?? 0,
    riskLevel: normalizeRiskLevel(annotations.riskLevel),
    approvalMode: normalizeApprovalMode(annotations.approvalMode),
    orchestrationMode: readString(annotations.orchestrationMode),
    raw: { tool }
  };
}
```

```ts
export async function listExpertCatalog(query?: ExpertCatalogQuery): Promise<ExpertCatalogPage> {
```

### 12.5 callCatalogSkill

```ts
export async function callCatalogSkill(
  input: CallCatalogSkillInput
): Promise<CallCatalogSkillResult> {
  const run = createExpertRun({
    runType: input.catalogKind === "expert_team" ? "team" : "single_expert",
    expertId: input.catalogKind === "expert" ? input.slug : undefined,
    teamId: input.catalogKind === "expert_team" ? input.slug : undefined,
    profileId: "remote",
    sessionId: input.sessionId,
    title: `${input.slug}/${input.skillName}`,
    userPrompt: input.prompt,
    status: "running",
    catalogSlug: input.slug,
    catalogKind: input.catalogKind,
    skillName: input.skillName,
    executionMode: "remote_mcp"
  });

  insertRunEvent({
    runId: run.runId,
    eventType: "mcp.call.started",
    payload: {
      slug: input.slug,
      catalogKind: input.catalogKind,
      skillName: input.skillName
    }
  });

  try {
    const result = await getExpertMcpClient().callSkill({
      slug: input.slug,
      skillName: input.skillName,
      arguments: {
        prompt: input.prompt,
        context: {
          source: "copilot-desktop",
          ...(input.context ?? {})
        }
      }
    });

    const responseText = extractTextContent(result);

    updateExpertRunStatus(run.runId, "completed", {
      responseText,
      structuredContentJson: result.structuredContent
        ? JSON.stringify(result.structuredContent)
        : undefined
    });

    insertRunEvent({
      runId: run.runId,
      eventType: "mcp.call.completed",
      payload: {
        slug: input.slug,
        catalogKind: input.catalogKind,
        skillName: input.skillName
      }
    });

    insertArtifact({
      runId: run.runId,
      profileId: "remote",
      title: `${input.slug} - ${input.skillName}`,
      artifactType: "markdown",
      previewText: responseText.slice(0, 4000),
      source: "expert_mcp_response"
    });

    return {
      ok: true,
      runId: run.runId,
      responseText,
      structuredContent: result.structuredContent
    };
  } catch (err) {
    const errorCode = isHermesExpertsError(err)
      ? err.code
      : "EXPERT_MCP_CALL_FAILED";
    const message = err instanceof Error ? err.message : String(err);

    updateExpertRunStatus(run.runId, "failed", {
      error: { code: errorCode, message }
    });

    insertRunEvent({
      runId: run.runId,
      eventType: "mcp.call.failed",
      payload: {
        slug: input.slug,
        catalogKind: input.catalogKind,
        skillName: input.skillName,
        errorCode,
        message
      }
    });

    return {
      ok: false,
      runId: run.runId,
      errorCode,
      message
    };
  }
}
```
